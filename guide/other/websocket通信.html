#websocket通信

#1. 资源更新页面

#1.1 选择资源

#
  // send to software
  {
    "code": "SELECT_ZY",            // 指令编码常量，设计软件根据code执行对应逻辑
    "param": {                      
      "zyId": "select_zyId"             // 传参： 双击选中的资源
    }
  }

  // send to web
  {
    "status": true,
    "code": "SELECT_ZY",
    "message": "string",
    "data": null
  }

#1.2 更新资源

#
  // send to software
  {
    "code": "UPDATE_COMPONENT_RESOURCE",
    "param": {
      zyId: "string",
      ids: "string"
    }
  }
  // send to web
  {
    "status": true,                       // 更新结果 true: 成功 false: 失败
    "code": "UPDATE_COMPONENT_RESOURCE",
    "message": "error message",           // 页面中会提示信息
    "data": null                          // 不需要返回值                    
  }

#2. WCC

#2.1 同步wcc数据

  code = "SYNC_DATA_WCC"
#
  // send to HomeDesign
  {
    "code": "SYNC_DATA_WCC",
    "param": {
      "name": string,
      "id":   string,
      "items": [
        {
          "Name":         string,           // 参数名称
          "Value":        string | number,  // 参数值
          "Descrip":      string,           // 参数描述
          "Slave":        string,           // 参数表达式
          "Type":         number,           // 参数类型 0: Double, 1：文本，2：布尔 3：整型
          "OutputFalg":   boolean,          // 在输出值的情况下 是否输出
          "OutputSlave":  string,           // 输出表达式（优先级高于输出值）
        },
        ...
      ]
    }
  }
  // response to web
  {
    "status": true,                       // 更新结果 true: 成功 false: 失败
    "code": "SYNC_DATA_WCC",
    "message": "error message",           // 页面中会提示信息
    "data": null                          // 不需要返回值                    
  }

#3. 约束设置

#3.1 同步约束数据

#
  // send to HomeDesign
  {
    "code": "SYNC_DATA_CONSTRAINT",
    "param": {
      "name": string,
      "code": string,
      "items": [
        {
          "id":       string,       // 约束ID
          "desc":     string,       // 约束说明
          "express":  string,       // 约束表达式
          "level":    number,       // 报错级别 - 0: 工艺报错  1：工艺警告  2：工艺审批
          "expandInfo": [           // 扩展信息 - 报价相关信息也填写到扩展信息中， 报价：type=quote
            { 
              "type":   string, 
              "value":  string 
            },
            ...
          ]
        },
        ...
      ]
    }
  }
// send to web
  {
    "status": true,                  // 更新结果 true: 成功 false: 失败
    "code": "SYNC_DATA_CONSTRAINT",
    "message": "error message",      // 页面中会提示信息
    "data": null                     // 不需要返回值                    
  }

#4. 参数导入

此结构通用参数导入功能，只有 code 字段和 items 字段的值会因为配置而不同。

#
  // send to HomeDesign
  {
    "code": "SYNC_DATA_{CONFIG}",    
    "param": {
      "name": string,
      "code": string,
      "items": [...]
    }
  }
// send to web
  {
    "status": true,                  // 更新结果 true: 成功 false: 失败
    "code": "SYNC_DATA_{CONFIG}",
    "message": "error message",      // 页面中会提示信息
    "data": null                     // 不需要返回值                    
  }

#4.1 配置：参数设置

参数设置 数据结构

code
  code = "SYNC_DATA_PARAMS"
items
  [{
    "baseSetting": {     // 基本设置
      "name": string,       // 名称
      "useto": string,      // 用途
      "desc": string,       // 描述
      "type":  1 | 2 | 3 | 4,   // 类型 - 1：数值 2：文本 3：布尔 4：整型
      "remark": string      // 备注
    }

    "valueSetting": {                       // 值设置
      "value": string | number,               //  值-值
      "value_express": string,                //  值-表达式
      "value_type": 'value' | 'express',      //  值类型 - value: 值, express: 表达式
      "min_value": number,                    //  最小值-值
      "min_express": string,                  //  最小值-表达式
      "min_type": 'value' | 'express',        //  最小值类型 - value: 值, express: 表达式
      "max_value": number,                    //  最大值-值
      "max_express": string,                  //  最大值-表达式
      "max_type": 'value' | 'express',        //  最大值类型 - value: 值, express: 表达式
      "f_display_value": boolean,             //  前端显示-值
      "f_display_express": string,            //  前端显示-表达式
      "f_display_type": 'value' | 'express',  //  前端显示类型 - value: 值, express: 表达式
      "refresh_express": string,              //  刷新条件表达式
      "f_readonly": boolean                   //  前端只读 
    },

    "seriesValue": {                  // 系列值
      "is_display": boolean,            // 前端是否显示 
      "is_series": boolean,             // 是否为系列值
      "series": [                     // ---系列值
        {                           
          "value": string,              //  选项值
          "is_display": boolean,        //  前端是否显示
          "display_value": string,      //  前端显示值
          "express": string,            //  表达式
          "remark": string              //  备注
        }, ...]

      "use": 'required' | 'priority',   //   优选/必选 - required: 必选， priority： 必选
      "express": string                 //  表达式
    },
    // 高级
    "advanced": {                       
      "is_mulit_rows": boolean,  // 多行显示
      "update_option": 'not_value' | 'value'  //  更新选项 not_value: 不更新值, value: 更新值
    }
  }, ...]
